Toggle


Toggle with otions in attribute



Usage

			
				AmsifyHelper.setBaseURL('jquery-plugins');
				$('.active').amsifyToggle({
					type : 'materialize',
					toggleClass: ['btn green', 'btn red'],
					toggleHTML: ['check', 'close'],
					flash: true,
					afterToggle: function() {
						console.info('After toggle');
					}
				});
				$('.block').amsifyToggle({
					type: 'materialize'
				});
