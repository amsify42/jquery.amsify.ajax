Toggle

Toggle with otions in attribute

Usage

			
				AmsifyHelper.setBaseURL('jquery-plugins');
				$('.active').amsifyToggle({
					toggleClass: ['btn-success', 'btn-danger'],
					toggleHTML: ['', ''],
					flash: true,
					afterToggle: function() {
						console.info('After toggle');
					}
				});
				$('.block').amsifyToggle();
